<script>
document.addEventListener("DOMContentLoaded", function() {
  // Adiciona um evento de clique para cada link na barra de navegação
  const links = document.querySelectorAll(".navbar a");
  links.forEach(link => {
    link.addEventListener("click", scrollToSection);
  });
});

function scrollToSection(event) {
  // Previne o comportamento padrão do link
  event.preventDefault();

  // Obtém o destino da âncora do atributo href
  const targetId = this.getAttribute("href");

  // Obtém o elemento de destino usando o ID
  const targetElement = document.querySelector(targetId);

  // Calcula a posição do elemento de destino
  const offsetTop = targetElement.offsetTop;

  // Rola a página suavemente até a posição do elemento de destino
  window.scrollTo({
    top: offsetTop,
    behavior: "smooth"
  });
}
</script>
